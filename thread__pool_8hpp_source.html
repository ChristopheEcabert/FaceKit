200 ThreadPool::ThreadPool(const std::size_t& size) : stop_(false) {
201  // Initialize workers
202  for (std::size_t i = 0; i < size; ++i) {
203  workers_.emplace_back([this](){
204  // Loop forever
205  while(true) {
206  // Get task
207  std::function<void()> task;
208  // Wait till some tasks are pending or if we stop
209  {
210  std::unique_lock<std::mutex> lock(this->cond_lock_);
211  this->cond_.wait(lock, [this](void) {
212  return this->stop_ || !this->queue_.empty();
213  });
214  // Stopping ?
215  if (this->stop_ && this->queue_.empty()) {
216  return;
217  }
218  // Queue not empty, pick pending job
219  task = std::move(this->queue_.top().second);
220  this->queue_.pop();
221  }
222  // Execute task
223  task();

245  std::forward<Args>(args)...));
246  // Define return object
