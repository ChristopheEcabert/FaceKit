194 ThreadPool::ThreadPool(const std::size_t& size) : stop_(false) {
195  // Initialize workers
196  for (std::size_t i; i < size; ++i) {
197  workers_.emplace_back([this](){
198  // Loop forever
199  while(true) {
200  // Get task
201  std::function<void()> task;
202  // Wait till some tasks are pending or if we stop
203  {
204  std::unique_lock<std::mutex> lock(this->lock_);
205  this->cond_.wait(lock, [this](void) {
206  return this->stop_ || !this->queue_.empty();
207  });
208  // Stopping ?

210  return;
211  }
212  // Queue not empty, pick pending job
213  task = std::move(this->queue_.top().second);
214  this->queue_.pop();
215  }

233 auto ThreadPool::Enqueue(const TaskPriority& priority, F&& f, Args&&... args)
234 -> std::future<typename std::result_of<F(Args...)>::type> {

236  using ret_type = typename std::result_of<F(Args...)>::type;
237 
238  // Create task

240  std::forward<Args>(args)...);
241  // Define return object
242  std::future<ret_type> res = task->get_future();
243  // Add to queue
244  {
245  std::unique_lock<std::mutex> lock(this->lock_);
246  if (this->stop_) {

248  }
249  this->queue_.emplace({priority, [task](void){ (*task)();}});
250  }
251  // Signal new job
