326  static bool IntersectObject(const Vector3<T>& p,
327  const Vector3<T>& dir,
328  const AABB<T>& bbox,
329  T* t) {
330  // Check for intersection
331  T t_enter = 0.f;
332  T t_exit = 1.f;
333  // For all three slabs
334  const T* dir_ptr = &dir.x_;
335  const T* p_ptr = &p.x_;
336  const T* bbox_min_ptr = &bbox.min_.x_;
337  const T* bbox_max_ptr = &bbox.max_.x_;
338  for (int i = 0; i < 3; ++i) {
339  if (std::abs(*dir_ptr) < 1e-9) {
340  // Ray is parallel to slab. No hit if origin not within slab
341  if ((*p_ptr < *bbox_min_ptr) || (*p_ptr > *bbox_max_ptr)) {
342  return false;
343  }
344  } else {
345  // Compute intersection t value of ray with near and far plane of slab
346  T ood = 1.0f / *dir_ptr;
347  T t_slab_enter = (*bbox_min_ptr - *p_ptr) * ood;
348  T t_slab_exit = (*bbox_max_ptr - *p_ptr) * ood;
349  // Make t_slab_enter be intersection with near plane, t_slab_exit with
350  // far plane
351  if (t_slab_enter > t_slab_exit) {
352  std::swap(t_slab_enter, t_slab_exit);
353  }
354  // make sure the slab interval and the current box intersection
355  // interval overlap
356  if (t_slab_enter > t_exit || t_slab_exit < t_enter) {
357  // nope. Ray missed the box.
358  return false;
359  } else {
360  // yep, the slab and current intersection interval overlap
361  // update the intersection interval
362  // Compute the intersection of slab intersection intervals
363  if (t_slab_enter > t_enter) { t_enter = t_slab_enter; }
364  if (t_slab_exit < t_exit) { t_exit = t_slab_exit; }
365  // Exit with no collision as soon as slab intersection becomes empty
366  if (t_enter > t_exit) { return false; }
367  }
368  }
369  // Inc ptr
370  ++dir_ptr;
371  ++p_ptr;
372  ++bbox_min_ptr;
373  ++bbox_max_ptr;
374  }
375  // Ray intersects all 3 slabs. Return point (q) and intersection t
376  // value (tmin)
377  *t = t_enter;
378  return true;
379  }
